一键安装

选择你的安装方式，30秒完成部署

bash 
# 一键安装主控端 (Dashboard)
curl -fsSL https://vstats.zsoft.cc/install.sh | sudo bash

# 或使用 wget
wget -qO- https://vstats.zsoft.cc/install.sh | sudo bash
bash 
# 在需要监控的服务器上运行
curl -fsSL https://vstats.zsoft.cc/agent.sh | sudo bash -s -- \
  --server http://YOUR_DASHBOARD_IP:3001 \
  --token "YOUR_TOKEN" \
  --name "$(hostname)"

# 可选参数: --location "US" --provider "Vultr"
bash 
# Docker 部署 (即将支持)
docker run -d \
  --name vstats \
  -p 3001:3001 \
  -v vstats-data:/data \
  ghcr.io/zsai001/vstats:latest
